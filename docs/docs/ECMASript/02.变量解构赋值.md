### 解构赋值
```
按照一定模式，从数组和对象中提取值，对变量进行赋值，这被称为解构赋值。

注意：频繁使用对象方法、数组元素，就可以使用解构赋值形式
```

### 数组解构
```js
    //数组解构
    const arr = ['张学友', '黎明', '刘德华', '郭富城']

    let [zhang, li, liu, guo] = arr
    console.log([zhang, li, liu, guo]) //['张学友', '黎明', '刘德华', '郭富城']

    let [zhang, li, liu,] = arr
    console.log([zhang, li, liu,]) //['张学友', '黎明', '刘德华']

    let [zhang = '张学友',b] = []
    console.log([zhang,b]) //['张学友', undefined]

    let [zhang ,...b] = arr
    console.log([zhang,b]) //['张学友', ['黎明', '刘德华', '郭富城']]
```

### 字符串解构
```js
    // 字符串解构
    let [a, b, c, d, e] = 'hello'
    console.log(a) // h
    console.log(b) // e
    console.log(c) // l
    console.log(d) // l
    console.log(e) // o
```

### 对象解构
```js
    //对象解构
    const lin = {
        name: '林志颖',
        tags: ['车手', '歌手', '小旋风', '演员']
    }
    let { name, tags } = lin
    console.log(name, tags) //林志颖 ['车手', '歌手', '小旋风', '演员']

    //复杂对象解构
    let wangfei = {
        name: '王菲',
        age: 18,
        songs: ['红豆', '流年', '暧昧', '传奇'],
        history: [
            { name: '窦唯' },
            { name: '李亚鹏' },
            { name: '谢霆锋' }
        ]
    }
    let { songs, history } = wangfei
    console.log(songs) //['红豆', '流年', '暧昧', '传奇']
    console.log(history)
```
