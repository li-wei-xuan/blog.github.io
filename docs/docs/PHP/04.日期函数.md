```php
// 设置时区 设置中国时区
date_default_timezone_set('PRC');

// 返回当前时间戳 单位：秒
echo time(); // 1676428777
echo "<br>";

// 时间戳转化为标准时间
echo date('Y-m-d H:i:s', time()); // 2023-02-15 10:39:37
echo "<br>";

// 标准时间转为时间戳
echo strtotime('2023-02-15 10:39:37'); // 1676428777
```

```php
// PHP获取当月第一天
echo date('Y-m-01'); // 2023-02-01
echo "<br>";

// PHP获取当月最后一天
echo date('Y-m-t'); // 2023-02-28
echo "<br>";

// PHP获取当前时间的前一天（前N天就是-N）
echo date('Y-m-d H:i:s', strtotime("-1 days"));
echo "<br>";

// PHP获取当前时间的后一天（后N天就是-N）
echo date('Y-m-d H:i:s', strtotime("+1 days"));
echo "<br>";

// 月
echo date('Y-m-d H:i:s', strtotime("-1 month"));
echo "<br>";

echo date('Y-m-d H:i:s', strtotime("+1 month"));
echo "<br>";

// 年
echo date('Y-m-d H:i:s', strtotime("-1 year"));
echo "<br>";

echo date('Y-m-d H:i:s', strtotime("+1 year"));
echo "<br>";

// 小时
echo date('Y-m-d H:i:s', strtotime("-1 hour"));
echo "<br>";

echo date('Y-m-d H:i:s', strtotime("+1 hour"));
echo "<br>";

// 分钟
echo date('Y-m-d H:i:s', strtotime("-1 minute"));
echo "<br>";

echo date('Y-m-d H:i:s', strtotime("+1 minute"));
echo "<br>";

// 秒
echo date('Y-m-d H:i:s', strtotime("-1 second"));
echo "<br>";

echo date('Y-m-d H:i:s', strtotime("+1 second"));
echo "<br>";

// 获取当月天数
echo date('t'); // 28
echo "<br>";
```
